\begin{corollary*}{}
Sejam $f, g \in \mathcal{M}^+(X, \mathcal{F})$ e $c \ge 0$. EntĂŁo $cf + g \in \mathcal{M}^+(X, \mathcal{F})$ e 
$$ \int (cf + g) \, d\mu = c \int f \, d\mu + \int g \, d\mu. $$
\end{corollary*}

\begin{proof}[\textbf{Dem:}]
O caso $c = 0$ Ă© imediato. Suponha $c > 0$ e $f, g \in \mathcal{M}^+(X, \mathcal{F})$. Pelo Teorema da primeira aula, existem sequĂȘncias $(\varphi_n)$ e $(\psi_n)$ de funĂ§Ă”es mensurĂĄveis simples, nĂŁo-decrescentes, que convergem para $f$ e $g$, respectivamente. 
Com isso, a sequĂȘncia $(c\varphi_n + \psi_n)$ tambĂ©m Ă© simples, nĂŁo-decrescente e converge pontualmente para $cf + g$. 
Desta forma, pela linearidade da integral de funĂ§Ă”es simples nĂŁo-negativas e pelo Teorema da ConvergĂȘncia MonĂłtona, obtemos
  \begin{align*}
  \int (cf + g) \, d\mu &= \int \lim_{n \to \infty} (c\varphi_n + \psi_n) \, d\mu = 
  \lim_{n \to \infty} \int (c\varphi_n + \psi_n) \, d\mu\\
  &=
  \lim_{n \to \infty} \left( c\int \varphi_n \, d\mu + \int \psi_n \, d\mu \right)
  = c\int f \, d\mu + \int g \, d\mu.   
  \end{align*}
\end{proof}
\bigline

A hipĂłtese da sequĂȘncia ser nĂŁo-decrescente Ă© essencial para o Teorema da ConvergĂȘncia MonĂłtona.
Considere em $(\mathbb{R}, \mathcal{M}, m)$ a sequĂȘncia dada por $f_n = \chi_{[n, n+1]}$. Assim, $$\lim f_n(x) = 0, \forall x \in \mathbb{R}.$$ Logo, $\int \lim f_n \, dm = 0$.
Por outro lado, $\int f_n \, dm = \int \chi_{[n, n+1]} \, dm = 1$. Logo, $\lim \int f_n \, dm = 1$. Portanto,
$$ \int \lim f_n \, dm \neq \lim \int f_n \, dm. $$
Na realidade Ă© sempre vĂĄlida uma desigualdade num contexto mais geral, como veremos a 
seguir.

\begin{lemma*}{ de Fatou}
Seja $(f_n)_{n \in \mathbb{N}}$ uma sequĂȘncia em $\mathcal{M}^+(X, \mathcal{F})$. EntĂŁo
$$ \int \liminf f_n \, d\mu \le \liminf \int f_n \, d\mu. $$
\end{lemma*}

\begin{proof}[\textbf{Dem:}]
Seja $g_m := \inf_{n \ge m} f_n$. JĂĄ vimos que $g_m \in \mathcal{M}^+(X, \mathcal{F})$ e, por construĂ§ĂŁo, $g_m \le g_{m+1}$, $\forall m \in \mathbb{N}$. 
AlĂ©m disso, $g_m \le f_n$, $\forall n \ge m$,

  $$\forall n \ge m, \int g_m \, d\mu \le \int f_n \, d\mu 
  \implies \int g_m \, d\mu \le \inf_{n \ge m} \int f_n \, d\mu.$$

Como $(g_m)$ Ă© nĂŁo-decrescente e $\lim g_m = \liminf f_n$, obtemos do Teorema da ConvergĂȘncia MonĂłtona que
$$ \int \liminf f_n \, d\mu = \int \lim g_m \, d\mu = \lim \int g_m \, d\mu \le \liminf \int f_n \, d\mu. $$
\end{proof}

\begin{corollary*}{}
Sejam $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida e $f \in \mathcal{M}^+(X, \mathcal{F})$. Temos que $f = 0$ $\mu$-q.t.p. se, e somente se, $\int f \, d\mu = 0$.
\end{corollary*}

\begin{proof}[\textbf{Dem:}]
($\Rightarrow$) Suponha que $f = 0$ $\mu$-q.t.p. Assim, existe $N \in \mathcal{F}$ tal que $f(x) = 0, \forall x \in N^c$ e $\mu(N) = 0$.
Desta forma, o conjunto $E = \{x \in X ; f(x) > 0\}$ satisfaz $E \subseteq N$, logo $\mu(E) = 0$.
Considere a sequĂȘncia $f_n = n \chi_E$. Como $f(x) \le \liminf f_n(x)$, $\forall x \in X$ (jĂĄ que em $E$, $f_n \to +\infty$ e em $E^c$, $f=0$), pelo Lema de Fatou concluĂ­mos que
$$ 0 \le \int f \, d\mu \le \int \liminf f_n \, d\mu \le \liminf \int f_n \, d\mu = \liminf (n \mu(E)) = 0. $$
Portanto $\int f \, d\mu = 0$.

($\Leftarrow$) Suponha que $\int f \, d\mu = 0$. Para cada $n \in \mathbb{N}$, temos pela Desigualdade de Chebyshev que o conjunto $E_{1/n} = \{x \in X ; f(x) \ge 1/n\}$ satisfaz
$$ 0 \le \mu(E_{1/n}) \le n \int f \, d\mu = 0 \implies \mu(E_{1/n}) = 0, \quad \forall n \in \mathbb{N}. $$
Como $\{x \in X ; f(x) > 0\} = \bigcup_{n \in \mathbb{N}} E_{1/n}$, segue que $\mu(\{f > 0\}) \le \sum \mu(E_{1/n}) = 0$. Portanto $f = 0$ $\mu$-q.t.p.
\end{proof}

\begin{corollary*}{}
Seja $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida e $(f_n)_{n \in \mathbb{N}}$ uma sequĂȘncia nĂŁo-decrescente em $\mathcal{M}^+(X, \mathcal{F})$ que converge pontualmente $\mu$-q.t.p. para $f \in \mathcal{M}^+(X, \mathcal{F})$. EntĂŁo $$\int f \, d\mu = \lim \int f_n \, d\mu.$$
\end{corollary*}

\begin{proof}[\textbf{Dem:}]
Seja $N \in \mathcal{F}$ tal que $\mu(N) = 0$ e $\lim f_n(x) = f(x)$, $\forall x \in N^c$. 
Assim, a sequĂȘncia $(f_n \chi_{N^c})$ Ă© uma sequĂȘncia nĂŁo-decrescente em $\mathcal{M}^+(X, \mathcal{F})$ que converge pontualmente para $f \chi_{N^c}$. 
Pelo Teorema da ConvergĂȘncia MonĂłtona, temos que
$$ \int f \chi_{N^c} \, d\mu = \lim \int f_n \chi_{N^c} \, d\mu. $$
Como $\mu(N) = 0$, as funĂ§Ă”es $f \chi_N$ e $f_n \chi_N$ sĂŁo nulas $\mu$-q.t.p. Logo, $\int f \chi_N \, d\mu = 0$ e $\int f_n \chi_N \, d\mu = 0$.
Desta forma, como $f = f \chi_N + f \chi_{N^c}$, concluĂ­mos que
\begin{align*}
    \int f \, d\mu &= \int (f \chi_N + f \chi_{N^c}) \, d\mu = \int f \chi_N \, d\mu + \int f \chi_{N^c} \, d\mu \\
    &= 0 + \lim \int f_n \chi_{N^c} \, d\mu = \lim \left( \int f_n \chi_N \, d\mu + \int f_n \chi_{N^c} \, d\mu \right) = \lim \int f_n \, d\mu.
\end{align*}
\end{proof}

\mysec{FunĂ§Ă”es IntegrĂĄveis}

Seja $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida. Vimos que uma funĂ§ĂŁo $f: X \to \overline{\mathbb{R}}$ Ă© mensurĂĄvel se, e somente se, $f^+$ e $f^-$ sĂŁo mensurĂĄveis, em que $f^+(x) = \max\{f(x), 0\}$ e $f^-(x) = \max\{-f(x), 0\}$.
AlĂ©m disso, vimos que $f = f^+ - f^-$ e $|f| = f^+ + f^-$. Assim, $f^+ = \frac{|f| + f}{2}$ e $f^- = \frac{|f| - f}{2}$. Note que $f^+$ e $f^-$ sĂŁo nĂŁo-negativas.

\begin{definition*}{}
Seja $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida e $f \in \mathcal{M}(X, \mathcal{F})$. Dizemos que $f$ Ă© \textbf{integrĂĄvel} se tanto $f^+$ quanto $f^-$ possuĂ­rem integrais finitas. Neste caso, definimos a integral de $f$ com respeito a $\mu$ por
$$ \int f \, d\mu := \int f^+ \, d\mu - \int f^- \, d\mu. $$
Se $E \in \mathcal{F}$, definimos $\int_E f \, d\mu := \int_E f^+ \, d\mu - \int_E f^- \, d\mu$. Denotaremos por $L^1(X, \mathcal{F}, \mu)$ o espaĂ§o das funĂ§Ă”es mensurĂĄveis integrĂĄveis.
\end{definition*}

Na definiĂ§ĂŁo acima, escrevemos $f$ como a diferenĂ§a entre $f^+$ e $f^-$, ambas nĂŁo-negativas e com integral finita. Vejamos que a integral de $f$ independe da decomposiĂ§ĂŁo de $f$.
Sejam $f_1, f_2$ funĂ§Ă”es mensurĂĄveis nĂŁo-negativas tais que suas integrais sĂŁo finitas e $f = f_1 - f_2$.
Como $f = f^+ - f^-$, temos que $f^+ - f^- = f_1 - f_2 \implies f^+ + f_2 = f_1 + f^-$. 
Como cada termo Ă© nĂŁo-negativo, pelos resultados vistos anteriormente temos que
$$ \int f^+ \, d\mu + \int f_2 \, d\mu = \int (f^+ + f_2) \, d\mu = \int (f_1 + f^-) \, d\mu = \int f_1 \, d\mu + \int f^- \, d\mu. $$
Agora, da finitude das integrais acima, obtemos $\int f^+ \, d\mu - \int f^- \, d\mu = \int f_1 \, d\mu - \int f_2 \, d\mu$.

\begin{theorem*}{}
Sejam $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida e $f \in \mathcal{M}(X, \mathcal{F})$. A funĂ§ĂŁo $f$ Ă© integrĂĄvel se, e somente se, $|f|$ Ă© integrĂĄvel. Neste caso, vale
$$ \left| \int f \, d\mu \right| \le \int |f| \, d\mu. $$
\end{theorem*}

\begin{proof}[\textbf{Dem:}]
($\Rightarrow$) Se $f$ Ă© integrĂĄvel, entĂŁo $f^+$ e $f^-$ possuem integral finita. Como $|f| = f^+ + f^-$, temos pela monotonicidade da integral para funĂ§Ă”es nĂŁo-negativas que
$$ \int |f| \, d\mu = \int (f^+ + f^-) \, d\mu = \int f^+ \, d\mu + \int f^- \, d\mu < +\infty. $$
Logo $|f|$ Ă© integrĂĄvel.

($\Leftarrow$) Suponha que $|f|$ seja integrĂĄvel. Como $f^+, f^-$ e $|f|$ sĂŁo nĂŁo-negativas, temos que $0 \le f^+ \le |f|$ e $0 \le f^- \le |f|$.
Assim, novamente pela monotonicidade da integral,
$$ 0 \le \int f^+ \, d\mu \le \int |f| \, d\mu < +\infty \quad \text{e} \quad 0 \le \int f^- \, d\mu \le \int |f| \, d\mu < +\infty. $$
Portanto $f$ Ă© integrĂĄvel.

\noindent
Por fim, vemos que
\begin{align*}
 \left| \int f \, d\mu \right| 
  &= \left| \int f^+ \, d\mu - \int f^- \, d\mu \right| 
  \le \left| \int f^+ \, d\mu \right| + \left| \int f^- \, d\mu \right|\\ 
  &= \int f^+ \, d\mu + \int f^- \, d\mu = \int (f^+ + f^-) \, d\mu = \int |f| \, d\mu. 
\end{align*}
\end{proof}

\begin{corollary*}{}
Sejam $f, g \in \mathcal{M}(X, \mathcal{F})$. Se $g$ Ă© integrĂĄvel e $|f| \le |g|$, entĂŁo $f$ Ă© integrĂĄvel e $\int |f| \, d\mu \le \int |g| \, d\mu$.
\end{corollary*}

\begin{proof}[\textbf{Dem:}]
Como $g$ Ă© integrĂĄvel, entĂŁo $|g|$ Ă© integrĂĄvel. Assim, com $|f| \le |g|$, temos que
$$ \int |f| \, d\mu \le \int |g| \, d\mu < +\infty. $$
Logo $|f|$ Ă© integrĂĄvel e, portanto, $f$ Ă© integrĂĄvel.
\end{proof}

\noindent
\textit{Obs:} O resultado mantĂ©m-se vĂĄlido se supormos $|f| \le |g|$ $\mu$-q.t.p. (Lista 3).

\begin{prop*}{}
Sejam $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida, $f, g \in L^1(X, \mathcal{F}, \mu)$ e $c \in \mathbb{R}$. EntĂŁo $cf$ e $f+g$ sĂŁo integrĂĄveis e vale
$$ \int cf \, d\mu = c \int f \, d\mu, \quad \int (f+g) \, d\mu = \int f \, d\mu + \int g \, d\mu. $$
\end{prop*}

\begin{proof}[\textbf{Dem:}]
Se $c = 0$, entĂŁo $cf = 0$ e, portanto, $\int cf \, d\mu = 0 = 0 \int f \, d\mu$.
Se $c > 0$:
$$ (cf)^+ = \frac{|cf| + cf}{2} = c \left( \frac{|f| + f}{2} \right) = c f^+ \quad \text{e} \quad (cf)^- = \frac{|cf| - cf}{2} = c \left( \frac{|f| - f}{2} \right) = c f^-. $$
Assim, pelas propriedades de integrais nĂŁo-negativas, $(cf)^+$ e $(cf)^-$ possuem integral finita e
\begin{align*}
  \int cf \, d\mu &= \int (cf)^+ \, d\mu - \int (cf)^- \, d\mu 
  = \int c f^+ \, d\mu - \int c f^- \, d\mu \\
  &= c \left( \int f^+ \, d\mu - \int f^- \, d\mu \right) = c \int f \, d\mu. 
\end{align*}

\noindent
Se $c < 0$:
$$ (cf)^+ = \frac{|cf| + cf}{2} = \frac{-c|f| + cf}{2} = -c \left( \frac{|f| - f}{2} \right) = -c f^-$$ 

\noindent
e
  $$\quad (cf)^- = \frac{|cf| - cf}{2} = \frac{-c|f| - cf}{2} = -c \left( \frac{|f| + f}{2} \right) = -c f^+. $$
Logo, $(cf)^+$ e $(cf)^-$ possuem integral finita e
\begin{align*}
  \int cf \, d\mu &= \int (cf)^+ \, d\mu - \int (cf)^- \, d\mu 
  = \int -cf^- \, d\mu - \int -cf^+ \, d\mu\\
  &= c \left( \int f^+ \, d\mu - \int f^- \, d\mu \right) = c \int f \, d\mu.  
\end{align*} 
Partimos agora para a soma. Como $f, g$ sĂŁo integrĂĄveis, entĂŁo $|f|, |g|$ sĂŁo integrĂĄveis. Como $|f+g| \le |f| + |g|$ e a integral de $|f| + |g|$ Ă© finita, segue do corolĂĄrio anterior que $f+g$ Ă© integrĂĄvel.

Por fim, temos que $f+g = (f^+ - f^-) + (g^+ - g^-) = (f^+ + g^+) - (f^- + g^-)$. 
  DaĂ­ obtemos que $(f+g)^+ + f^- + g^- = (f+g)^- + f^+ + g^+$.
  Como todas as funĂ§Ă”es sĂŁo nĂŁo-negativas, segue que  
\begin{align*}
  \int (f+g) \, d\mu &=  \int (f+g)^+ \, d\mu - \int (f+g)^- \, d\mu\\
    &= \int f^+ \, d\mu  + \int g^+ \, d\mu - \int f^- \, d\mu - \int g^- \, d\mu \\
     &= \int f \, d\mu + \int g \, d\mu.
\end{align*}
\end{proof}

\begin{theorem*}{ da ConvergĂȘncia Dominada}
Sejam $(X, \mathcal{F}, \mu)$ um espaĂ§o de medida e $(f_n)$ uma sequĂȘncia de funĂ§Ă”es em $L^1(X, \mathcal{F}, \mu)$ que converge $\mu$-q.t.p. para uma funĂ§ĂŁo mensurĂĄvel $f$. Se existir uma funĂ§ĂŁo integrĂĄvel $g$ tal que $|f_n| \le g$ $\mu$-q.t.p., $\forall n \in \mathbb{N}$, entĂŁo $f$ Ă© integrĂĄvel e 
$$ \int f \, d\mu = \lim_{n \to \infty} \int f_n \, d\mu. $$
\end{theorem*}

\begin{proof}[\textbf{Dem:}]
Redefinindo $f_n, g$ e $f$ num conjunto de medida nula, podemos supor sem perda de generalidade que $f_n$ converge pontualmente para $f$ e $|f_n| \le g$, $\forall x \in X$. (Lista 3).

Como $f_n$ converge para $f$ e $|f_n| \le g$, $\forall n \in \mathbb{N}$, entĂŁo $|f| \le g$. Assim, tanto $f_n$ quanto $f$ sĂŁo integrĂĄveis, $\forall n \in \mathbb{N}$. 
Para cada $n \in \mathbb{N}$, temos que $g - f_n$ e $g + f_n$ sĂŁo mensurĂĄveis nĂŁo-negativas. Pelo Lema de Fatou, temos que
\begin{align*}
  \int g \, d\mu - \int f \, d\mu &= \int (g - f) \, d\mu = \int \liminf (g - f_n) \, d\mu\\ 
  &\le \liminf \int (g - f_n) \, d\mu = \int g \, d\mu - \limsup \int f_n \, d\mu. 
\end{align*}
TambĂ©m,
  \begin{align*}
  \int g \, d\mu + \int f \, d\mu &= \int (g + f) \, d\mu = \int \liminf (g + f_n) \, d\mu\\
  &\le \liminf \int (g + f_n) \, d\mu = \int g \, d\mu + \liminf \int f_n \, d\mu.  
  \end{align*}
Como $g$ Ă© integrĂĄvel, podemos subtrair $\int g \, d\mu$ das desigualdades, obtendo:
$$ \limsup \int f_n \, d\mu \le \int f \, d\mu \le \liminf \int f_n \, d\mu. $$
Portanto, existe o limite de $\int f_n \, d\mu$ e $\lim_{n \to \infty} \int f_n \, d\mu = \int f \, d\mu$.
\end{proof}
